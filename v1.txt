local Example = function(Tab)
	local left = Tab:DrawSection({
		Name = "Example",
		Position = "left"
	})
	
	local right = Tab:DrawSection({
		Name = "Example",
		Position = "right"
	})
	
	local load = function(section)
		section:AddToggle({
			Name = "Toggle",
			Callback = print
		}).Link:AddKeybind({
			Name = "Keybind",
			Default = nil,
			Callback = print
		});
		
		section:AddSlider({
			Name = "Slider",
			Min = 0,
			Max = 100,
			Round = 1,
			Type = "%",
			Default = 50,
			Callback = print
		})
		
		section:AddDropdown({
			Name = "Single Dropdown",
			Values = {"Head","Chest","Pelvis","Arms","Legs"},
			Default = "Head",
			Multi = false,
			Callback = print
		})
		
		section:AddDropdown({
			Name = "Multi Dropdown",
			Values = {"Head","Chest","Pelvis","Arms","Legs"},
			Default = {"Head"},
			Multi = true,
			Callback = print
		})
		
		section:AddColorPicker({
			Name = "ColorPicker",
			Default = Color3.fromRGB(107, 255, 28),
			Flag = tostring(math.random(1,1000))..tostring(math.random(1,1000))..tostring(math.random(1,1000))..tostring(math.random(1,1000)),
			Callback = print
		});
		
		section:AddKeybind({
			Name = "Keybind",
			Default = nil,
			Callback = print
		});
		
		section:AddColorPicker({
			Name = "Double ColorPicker",
			Default = Color3.fromRGB(107, 255, 28),
			Callback = print
		}).Link:AddColorPicker({
			Name = "Double ColorPicker",
			Default = Color3.fromRGB(255, 0, 4),
			Callback = print
		});
	end;
	
	load(left)
	load(right)
end;

	Default = 0.1,
	Flag = "hngrh13241",

		_G.RAINBOW_SPEED = v;
	end,
})

task.spawn(function()
	local x = 0

	while true do task.wait(_G.RAINBOW_SPEED or 0.1)
		
		if _G.RAINBOW then
			Compkiller:ChangeHighlightColor(Color3.fromHSV(x,1,1))

			x = x + 2/255

			if x >= 1 then
				x = 0
			end
		end;
	end
end)
